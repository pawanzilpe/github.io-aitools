Here's a quick Python snippet I found useful for data cleaning:
python_code.py
import pandas as pd

def clean_data(df):
    # Fill missing values
    df.fillna(method='ffill', inplace=True)
    
    # Remove duplicates
    df.drop_duplicates(inplace=True)
    
    # Convert to proper data types
    df['date'] = pd.to_datetime(df['date'])
    df['price'] = pd.to_numeric(df['price'])
    
    return df
