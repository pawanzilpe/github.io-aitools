// Sample JavaScript for job card functionality
document.querySelectorAll('.action-btn').forEach(button => {
  button.addEventListener('click', function() {
    const action = this.classList.contains('apply') ? 
                  'apply' : 'save';
    
    if (action === 'apply') {
      // Redirect to application form
      window.location.href = '/apply';
    } else {
      // Save job to favorites
      const jobId = this.closest('.job-card').dataset.id;
      saveJobToFavorites(jobId);
    }
  });
});

function saveJobToFavorites(jobId) {
  // API call to save job
  fetch(`/api/jobs/${jobId}/save`, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Authorization': `Bearer ${userToken}`
    }
  })
  .then(response => response.json())
  .then(data => {
    if (data.success) {
      showNotification('Job saved successfully!');
    }
  });
}

// Job model (MongoDB schema)
const mongoose = require('mongoose');

const jobSchema = new mongoose.Schema({
  title: {
    type: String,
    required: true
  },
  company: {
    type: String,
    required: true
  },
  location: String,
  salary: String,
  jobType: {
    type: String,
    enum: ['Full-time', 'Part-time', 'Contract', 'Internship']
  },
  description: String,
  requirements: [String],
  createdBy: {
    type: mongoose.Schema.Types.ObjectId,
    ref: 'User'
  },
  createdAt: {
    type: Date,
    default: Date.now
  }
});

module.exports = mongoose.model('Job', jobSchema);
// Job controller
exports.createJob = async (req, res) => {
  try {
    const { title, company, location, salary, jobType, description, requirements } = req.body;
    
    const job = new Job({
      title,
      company,
      location,
      salary,
      jobType,
      description,
      requirements: requirements.split('\n'),
      createdBy: req.user.id
    });
    
    await job.save();
    
    res.status(201).json({
      success: true,
      data: job
    });
  } catch (err) {
    res.status(500).json({
      success: false,
      error: 'Server error'
    });
  }
};

exports.getJobs = async (req, res) => {
  try {
    const jobs = await Job.find().sort({ createdAt: -1 });
    res.json({
      success: true,
      count: jobs.length,
      data: jobs
    });
  } catch (err) {
    res.status(500).json({
      success: false,
      error: 'Server error'
    });
  }
};
// Job routes
const express = require('express');
const router = express.Router();
const { protect } = require('../middleware/auth');
const {
  createJob,
  getJobs,
  getJob,
  updateJob,
  deleteJob
} = require('../controllers/jobs');

// Public routes
router.route('/')
  .get(getJobs);

router.route('/:id')
  .get(getJob);

// Protected routes (require authentication)
router.use(protect);

router.route('/')
  .post(createJob);

router.route('/:id')
  .put(updateJob)
  .delete(deleteJob);

module.exports = router;
