// JavaScript for Form Validation
document.getElementById('registrationForm').addEventListener('submit', function(e) {
  e.preventDefault();
  
  // Get form values
  const fullName = document.getElementById('fullName').value;
  const email = document.getElementById('email').value;
  const phone = document.getElementById('phone').value;
  const password = document.getElementById('password').value;
  const confirmPassword = document.getElementById('confirmPassword').value;
  const country = document.getElementById('country').value;
  
  // Simple validation
  if (password !== confirmPassword) {
    alert('Passwords do not match!');
    return;
  }
  
  if (!validateEmail(email)) {
    alert('Please enter a valid email address!');
    return;
  }
  
  if (country === '') {
    alert('Please select your country!');
    return;
  }
  
  // If validation passes
  alert(`Registration successful!\nWelcome ${fullName}`);
  this.reset();
});

// Email validation function
function validateEmail(email) {
  const re = /^(([^<>()\[\]\\.,;:\s@"]+(\.[^<>()\[\]\\.,;:\s@"]+)*)|(".+"))@((\[[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}])|(([a-zA-Z\-0-9]+\.)+[a-zA-Z]{2,}))$/;
  return re.test(String(email).toLowerCase());
}

// Tab switching functionality
document.querySelectorAll('.tab').forEach(tab => {
  tab.addEventListener('click', () => {
    // Remove active class from all tabs
    document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
    
    // Add active class to clicked tab
    tab.classList.add('active');
    
    // Hide all code sections
    document.querySelectorAll('.html-code, .css-code, .js-code').forEach(code => {
      code.classList.remove('active');
    });
    
    // Show the selected code section
    const tabType = tab.getAttribute('data-tab');
    document.querySelector(`.${tabType}-code`).classList.add('active');
  });
});
