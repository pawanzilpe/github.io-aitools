    shareBtn.addEventListener('click', function() {
        if (navigator.share) {
            navigator.share({

                text: 'Check out this job opportunity!',
                url: window.location.href
            })

            alert('Web Share API not supported in your browser. You can copy the URL manually.');
        }
    });
    
    document.querySelector('.apply-section').appendChild(shareBtn);
});

// Simple Node.js backend for job vacancy application
const express = require('express');
const bodyParser = require('body-parser');
const app = express();
const port = 3000;

// Middleware to parse JSON and form data
app.use(bodyParser.json());
app.use(bodyParser.urlencoded({ extended: true }));

// Serve static files (HTML, CSS, JS)
app.use(express.static('public'));

// In-memory database for job applications
let applications = [];

// API endpoint to submit job application
app.post('/apply', (req, res) => {
    const { jobId, name, email, resume } = req.body;
    
    if (!name || !email) {
        return res.status(400).json({ error: 'Name and email are required' });
    }
    
    // Create application object
    const application = {
        id: applications.length + 1,
        jobId,
        name,
        email,
        resume: resume || null,
        date: new Date().toISOString(),
        status: 'Submitted'
    };
    
    // Add to applications array
    applications.push(application);
    
    // Log the application
    console.log('New application received:', application);
    
    // Send success response
    res.json({ 
        success: true, 
        message: 'Application submitted successfully!', 
        applicationId: application.id 
    });
});

// API endpoint to get job details
app.get('/job/:id', (req, res) => {
    const jobId = req.params.id;
    
    // In a real app, this would come from a database
    const job = {
        id: jobId,
        title: 'Senior Web Developer',
        company: 'Tech Innovations Ltd.',
        location: 'Remote / New York, NY',
        salary: '$90,000 - $120,000 per year',
        type: 'Contract',
        description: 'We are seeking a talented Senior Web Developer...',
        requirements: [
            '5+ years of experience in web development',
            'Strong proficiency in JavaScript',
            'Experience with React.js',
            'Familiarity with RESTful APIs'
        ]
    };
    
    res.json(job);
});

// Start the server
app.listen(port, () => {
    console.log(`Server running at http://localhost:${port}`);
});

module.exports = app;
